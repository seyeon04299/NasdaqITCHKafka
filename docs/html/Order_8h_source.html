244 inline void
245 Order::CloneOrderComboLegs(OrderComboLegListSPtr& dst, const OrderComboLegListSPtr& src)
246 {
247  if (!src.get())
248  return;
249 
250  dst->reserve(src->size());
251 
252  OrderComboLegList::const_iterator iter = src->begin();
253  const OrderComboLegList::const_iterator iterEnd = src->end();
254 
255  for (; iter != iterEnd; ++iter) {
256  const OrderComboLeg* leg = iter->get();
257  if (!leg)
258  continue;
259  dst->push_back(OrderComboLegSPtr(new OrderComboLeg(*leg)));
260  }
261 }
