82 HEADER_FUNCTION size_t fsize_orDie(const char *filename)
83 {
84  struct stat st;
85  if (stat(filename, &st) != 0)
86  {
87  /* error */
88  perror(filename);
89  exit(ERROR_fsize);
90  }
91 
92  off_t const fileSize = st.st_size;
93  size_t const size = (size_t)fileSize;
94  /* 1. fileSize should be non-negative,
95  * 2. if off_t -> size_t type conversion results in discrepancy,
96  * the file size is too large for type size_t.
97  */
98  if ((fileSize < 0) || (fileSize != (off_t)size))
99  {
100  fprintf(stderr, "%s : filesize too large \n", filename);
101  exit(ERROR_largeFile);
102  }
103  return size;
104 }

201 HEADER_FUNCTION size_t loadFile_orDie(const char *fileName, void *buffer, size_t bufferSize)
202 {
203  size_t const fileSize = fsize_orDie(fileName);
204  CHECK(fileSize <= bufferSize, "File too large!");
205 
206  FILE *const inFile = fopen_orDie(fileName, "rb");
207  size_t const readSize = fread(buffer, 1, fileSize, inFile);
208  if (readSize != (size_t)fileSize)
209  {
210  fprintf(stderr, "fread: %s : %s \n", fileName, strerror(errno));
211  exit(ERROR_fread);
212  }
213  fclose(inFile); /* can't fail, read only */
214  return fileSize;
215 }
