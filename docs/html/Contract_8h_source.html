170 inline void Contract::CloneComboLegs(ComboLegListSPtr &dst, const ComboLegListSPtr &src)
171 {
172  if (!src.get())
173  return;
174 
175  dst->reserve(src->size());
176 
177  ComboLegList::const_iterator iter = src->begin();
178  const ComboLegList::const_iterator iterEnd = src->end();
179 
180  for (; iter != iterEnd; ++iter)
181  {
182  const ComboLeg *leg = iter->get();
183  if (!leg)
184  continue;
185  dst->push_back(ComboLegSPtr(new ComboLeg(*leg)));
186  }
187 }
